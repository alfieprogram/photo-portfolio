      const texts = ["Welcome to my Photo Portfolio!", "I am a Photographer that use Iphone 11 and Nikon D3100", "Time for your photo ðŸ“¸"]; // Replace with your array of h1 elements
      let index = 0;
      const textElement = document.getElementById("auto-type-text");
  
      function autoType() {
        if (index < texts.length) {
          const currentText = texts[index];
          const currentLength = currentText.length;
  
          textElement.textContent = currentText.substring(0, currentLength);
          index++;
  
          setTimeout(autoType, 10000); // Adjust the delay (in milliseconds) between each character
        } else {
          // Reset index for continuous looping
          index = 0;
          setTimeout(autoType, 1000); // Adjust the delay before restarting (in milliseconds)
        }
      }
  
      // Start the auto-typing when the page loads
      window.onload = autoType;
    </script>
